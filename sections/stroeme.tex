\textbf{Stromdichte}:
\[
	j \approx \frac{\textrm{Strom}}{\textrm{Fläche}} \approx \vec{E}
\]\[
	j = \rho \cdot E = \frac{1}{\rho}E
\]

\textbf{Elektrische Leitfähigkeit}:
\[
	j = \sigma \cdot \vec{E}
\]\[
	\sigma = \frac{j}{\vec{E}}
\]

\textbf{Spezifischer Widerstand}:
\[
	\rho = \frac{1}{\sigma}
\]
